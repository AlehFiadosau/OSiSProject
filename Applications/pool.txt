        iters = n / processesNumber
        h = (self.Yf - self.Ys) / processesNumber

        p = Pool(processes=processesNumber)
        for index in range(processesNumber):
            ys = self.Ys + h * index
            yf = self.Ys + h * (index + 1)
            temp = p.apply_async(self.calcFromY, (ys, yf, math.ceil(iters)))
            result += temp.get()

        p.close()
        p.join()
